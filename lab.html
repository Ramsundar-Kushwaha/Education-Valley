Lab 01
#include <iostream>
using namespace std;

int main()
{
    float percentage;
    cout << "Enter the percentage of the student: ";
    cin >> percentage;

    if(percentage >= 90)
    {
      cout << "Grade A+";
    }

    else if((percentage < 90) && (percentage >= 70))
    {
      cout << "Grade A";
    }

    else if((percentage < 70) && (percentage >= 60))
    {
      cout << "Grade B";
    }

    else if((percentage < 60) && (percentage >= 40))
    {
      cout << "Grade C";
    }

    else if((percentage < 40) && (percentage >= 30))
    {
      cout << "Grade D";
    }

    else
    {
      cout << "Failed";
    }
}

#include <iostream>
using namespace std;

class student
{
  private:
    string name;
    int rollNo;
    char grade;

  public:
    void readDetails()
    {
      cout << "Enter Name: ";
      cin >> name;
      cout << endl;

      cout << "Enter Roll No: ";
      cin >> rollNo;
      cout << endl;

      cout << "Enter Grade: ";
      cin >> grade;
      cout << endl;
    }

    void displayResult()
    {
      cout << "Name: " << name << " Roll No: " << rollNo << " Grade: " << grade << endl;
    }
};

int main()
{
  const int numberOfTheStudent = 3;
  student stu1[numberOfTheStudent];

  cout << "Enter the details of " << numberOfTheStudent << " students: " << endl;

  for(int i = 0; i < numberOfTheStudent; i++)
  {
    cout << "student" << i + 1 << ";" << endl;
    stu1[i].readDetails();
  }
}
